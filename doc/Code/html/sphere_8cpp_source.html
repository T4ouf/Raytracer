30 Hit Sphere::intersect(const Ray &ray)
31 {
33  //First we look for the intersection point
34 
35  //t is the distance from ray's origin to intersection point
36  double t = 0;
37 
38  //Vector from ray's origin to center of the sphere
39  Vector L = (position - ray.O);
40 
41  //We project L on the ray's direction (and get the total length)
42  double tca = L.dot(ray.D);
43 
44  //if the sphere is behind the ray => no hit point
45  if (tca < 0) {
46  return Hit::NO_HIT();
47  }
48 
49  //d is the distance between the sphere's origin and the "inbound part" of the ray (perpendicularly)
50  double d = sqrt( L.length_2() - (tca * tca));
51 
52  if (d < 0) {
53  return Hit::NO_HIT();
54  }
55 
56  //Normally impossible (but remove some NaN values of thc so we keep it)
57  if (d > radius) {
58  return Hit::NO_HIT();
59  }
60 
61  //length of the ray that is "inside" the sphere (trigo computation)
62  double thc = sqrt(radius * radius - d * d);
63 
64  //from the total length we remove the part inside the sphere => we get the length until intersection
65  t = tca - thc;
66  //We also compute our vector from ray's origin to the impact point
67  Vector P = (ray.O + (t * ray.D));
68 
70  //Then we compute the normal at this point
71 
72  //Vector from sphere center to impact point (normal to the surface)
73  Vector CP = P - position;
74  Vector N = (CP).normalized();
75 
76  return Hit(t,N);
77 }
