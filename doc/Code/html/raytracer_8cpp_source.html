171 Object* Raytracer::parseObject(const YAML::Node& node)
172 {
173  Object *returnObject = NULL;
174  std::string objectType;
175  node["type"] >> objectType; //We parse the type of object (what kind of object is it ?)
176 
177  // Parse a Sphere (3D object)
178  if (objectType == "sphere") {
179  Point pos;
180  node["position"] >> pos; //parse its position
181  double r;
182  node["radius"] >> r; //parse its radius
183 
184  //Small trick to manage rotation for spheres
185  auto upValue = node.FindValue("rotation-axis");
186  Vector up = Vector(0, 1, 0);
187  if (upValue != NULL) {
188  up = parseTriple(*upValue).normalized();
189  }
190 
191  double rot = 0.0f;
192  auto angleValue = node.FindValue("angle");
193  if (angleValue != NULL) {
194  *angleValue >> rot;
195  }
196 
197  //construct and return the sphere
198  Sphere *sphere = new Sphere(pos,r,rot,up);
199  returnObject = sphere;
200  }
201  // Parse a Triangle (2D object)
202  else if (objectType == "triangle") {
203  //parse all points of the triangle
204  Point p1;
205  Point p2;
206  Point p3;
207  node["p1"] >> p1;
208  node["p2"] >> p2;
209  node["p3"] >> p3;
210 
211  //construct and return the triangle
212  Triangle* triangle = new Triangle(p1,p2,p3);
213  returnObject = triangle;
214  }
215  // Parse a Quad (2D object)
216  else if (objectType == "quad") {
217  //parse all points of the quad
218  Point p1;
219  Point p2;
220  Point p3;
221  Point p4;
222  node["p1"] >> p1;
223  node["p2"] >> p2;
224  node["p3"] >> p3;
225  node["p4"] >> p4;
226 
227  //construct and return the triangle
228  Quad* quad = new Quad(p1, p2, p3, p4);
229  returnObject = quad;
230  }
231  // Parse a Plane (2D infinite object)
232  else if (objectType == "plane") {
233  //parse all the points defining the plane in 3D space
234  Point p1;
235  Point p2;
236  Point p3;
237  node["p1"] >> p1;
238  node["p2"] >> p2;
239  node["p3"] >> p3;
240 
241  //Small trick to manage rotation for planes (textures)
242  auto upValue = node.FindValue("rotation-axis");
243  Vector up = Vector(0, 1, 0);
244  if (upValue != NULL) {
245  up = parseTriple(*upValue).normalized();
246  }
247 
248  double rot = 0;
249  auto angleValue = node.FindValue("angle");
250  if (angleValue != NULL) {
251  *angleValue >> rot;
252  }
253 
254  //construct and return the plane
255  Plane* plane = new Plane(p1, p2, p3, up, rot);
256  returnObject = plane;
257  }
258  // Parse a 3D Model (3D object from file)
259  else if(objectType == "model"){
260 
261  //Load the model from .obj files using glm
262  std::string path;
263  node["path"] >> path;
264  char* path2 = &path[0];
265  auto model = glmReadOBJ(path2);
266 
267  double scaleFactor;
268  node["scaleFactor"] >> scaleFactor;
269 
270  //More visual => x10
271  scaleFactor = scaleFactor * 10;
272 
273  //We properly scale our model
274  glmUnitize(model);
275  glmScale(model, float(scaleFactor));
276 
277  Point position = parseTriple(node["position"]);
278 
279  //Array to store all the vertices
280  double* dblArray = new double[model->numtriangles * 9];
281 
282  //We get the dimensions of our model
283  float dim[3];
284  glmDimensions(model, dim);
285 
286 
287  double r = 0.0; //radius of the bounding sphere (minimize computation for ray's hit detection with an easier "global" hitbox)
288  if (dim[0] >= dim[1] && dim[0] >= dim[2]) {
289  r = dim[0] * scaleFactor;
290  }
291  else if (dim[1] >= dim[0] && dim[1] >= dim[2]) {
292  r = dim[1] * scaleFactor;
293  }
294  else if (dim[2] >= dim[1] && dim[2] >= dim[0]) {
295  r = dim[2] * scaleFactor;
296  }
297 
298  //we create a blank model object...
299  Model* m = new Model(position, r, 0, Vector(0, 1, 0));
300 
301  //we parse all the triangles
302  for (size_t i = 0; i < model->numtriangles; i++) {
303 
304  for (size_t j = 0; j < 3; j++){
305  //cout << model->triangles[i].vindices[j] << '\t';
306 
307  for (size_t k = 0; k < 3; k++){
308 
309  dblArray[9 * i + 3 * j + k] = model->vertices[3*model->triangles[i].vindices[j]+k];
310  }
311 
312  }
313  //cout << '\n';
314 
315  Point p1 = Point(dblArray[9 * i] , dblArray[9 * i + 1], dblArray[9 * i + 2]) + position;
316  Point p2 = Point(dblArray[9 * i + 3], dblArray[9 * i + 4], dblArray[9 * i + 5]) + position;
317  Point p3 = Point(dblArray[9 * i + 6], dblArray[9 * i + 7], dblArray[9 * i + 8]) + position;
318 
319  Triangle t = Triangle(p1, p2, p3);
320  m->meshes.push_back(t);
321  }
322 
323  /* Debug print
324  cout << "\n-------------------------------------------------------\n";
325 
326  for (int i2 = 0; i2 < model->numtriangles * 9; i2++) {
327  cout << dblArray[i2] << '\t';
328 
329  if ((i2+1) % 3 == 0) {
330  cout << '\n';
331  }
332 
333  }
334  cout << '\n';*/
335 
336  //return the 3D model
337  returnObject = m;
338 
339  //clean memory
340  glmDelete(model);
341  delete[] dblArray;
342 
343  }
344 
345  //If we found an object to parse...
346  if (returnObject != NULL) {
347  //... we read the material and attach to object
348  returnObject->material = parseMaterial(node["material"]);
349  }
350 
351  return returnObject;
352 }

405 Camera* Raytracer::parseCamera(const YAML::Node& node) {
406  Camera* c = new Camera();
407 
408  // Read scene configuration options
409  c->eye = parseTriple(node["eye"]);
410  c->c = parseTriple(node["center"]);
411  Vector upInit = parseTriple(node["up"]);
412  Vector CEye = (c->c - c->eye).normalized();
413  Vector side = CEye.cross(upInit);
414  c->side = side;
415  c->up = side.cross(CEye);
416 
417  double width = node["viewSize"][0];
418  double height = node["viewSize"][1];
419  c->aspectRatio = width / height;
420  c->baseline = height; //define the baseline as the height => we will multiply the width with the AspectRatio
421 
422  return c;
423 }

427 * Method that reads a full scene from a file
428 * @param[in] inputFilename, the path to the file
429 * @return a boolean stating if the method succeeded or not
430 */
431 bool Raytracer::readScene(const std::string& inputFilename)
432 {
433  // Initialize a new scene
434  scene = new Scene();
435 
436  // Open file stream for reading and have the YAML module parse it
437  std::ifstream fin(inputFilename.c_str());
438  if (!fin) {
439  std::cerr << "Error: unable to open " << inputFilename << " for reading." << std::endl;;
440  return false;
441  }
442  try {
443  YAML::Parser parser(fin);
444 
445  //if the parser was correctly created...
446  if (parser) {
447 
448  //we check the file for the first node
449  YAML::Node doc;
450  parser.GetNextDocument(doc);
451 
452  //check for the raytracing type
453  auto renderType = doc.FindValue("RaytracingType");
454  if (renderType != NULL) {
455  scene->setRaytracingType(parseType(*renderType));
456  }
457  else {
458  scene->setRaytracingType(raytracingType::PHONG); //default rendering mode
459  }
460 
461  //check for the camera param
462  scene->setCamera(parseCamera(doc["Camera"]));
463 
464  //check for the shadow boolean
465  auto shadowPresence = doc.FindValue("Shadows");
466  if (shadowPresence != NULL) {
467  scene->setShadowBool(parseType(*shadowPresence));
468  }
469  else {
470  scene->setShadowBool(false);
471  }
472 
473  //check for the recursion depth
474  auto recursionDepth = doc.FindValue("MaxRecursionDepth");
475  if (recursionDepth != NULL) {
476  scene->setMaxRecursion(parseReflectionDepth(*recursionDepth));
477  }
478  else {
479  scene->setMaxRecursion(0);
480  }
481 
482  //check for the super sampling factor
483  auto superSampling = doc.FindValue("SuperSampling");
484  if (superSampling != NULL) {
485  scene->camera.superSampling = parseSuperSampling(*superSampling);
486  }
487  else {
488  scene->camera.superSampling = 1;
489  }
490 
491  //check for the Gooch parameters
492  auto GoochParameters = doc.FindValue("GoochParameters");
493  if (GoochParameters != NULL) {
494  scene->setGoochParams(parseGooch(*GoochParameters));
495  }
496  else {
497  scene->setGoochParams({ 0,0,1,1 });
498  }
499 
500 
501  // Read and parse the scene objects
502  const YAML::Node& sceneObjects = doc["Objects"];
503  if (sceneObjects.GetType() != YAML::CT_SEQUENCE) {
504  std::cerr << "Error: expected a sequence of objects." << std::endl;
505  return false;
506  }
507  for(YAML::Iterator it=sceneObjects.begin();it!=sceneObjects.end();++it) {
508  Object *obj = parseObject(*it);
509 
510  // Only add object to the scene if it is recognized
511  if (obj) {
512  scene->addObject(obj);
513  } else {
514  std::cerr << "Warning: found object of unknown type, ignored." << std::endl;
515  }
516  }
517 
518  // Read and parse light definitions
519  const YAML::Node& sceneLights = doc["Lights"];
520  if (sceneObjects.GetType() != YAML::CT_SEQUENCE) {
521  std::cerr << "Error: expected a sequence of lights." << std::endl;
522  return false;
523  }
524  for(YAML::Iterator it=sceneLights.begin();it!=sceneLights.end();++it) {
525  scene->addLight(parseLight(*it));
526  }
527  }
528  if (parser) {
529  std::cerr << "Warning: unexpected YAML document, ignored." << std::endl;
530  }
531  } catch(YAML::ParserException& e) {
532  std::cerr << "Error at line " << e.mark.line + 1 << ", col " << e.mark.column + 1 << ": " << e.msg << std::endl;
533  return false;
534  }
535 
536  //print result for information
537  std::cout << "YAML parsing results: " << scene->getNumObjects() << " objects read." << std::endl;
538  return true;
539 }
