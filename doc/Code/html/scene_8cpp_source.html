475 void Scene::render(Image& img){
476 
477  //dimensions of the image
478  long long int w = img.width();
479  long long int h = img.height();
480 
481  //dimensions of a pixel
482  Vector pixelWidth = camera.side;
483  Vector pixelHeight = -camera.up;
484 
485  //MidPoint of the image
486  Vector M = camera.c - pixelWidth * double(w) / 2.0f - pixelHeight * double(h) / 2.0f;
487 
488  //for each pixel of the image
489  #pragma omp parallel for
490  for (long long int y = 0; y < h; y++) {
491  for (long long int x = 0; x < w; x++) {
492 
493  //We are going to sample a pixel color by tracing multiple rays (SSAA)
494  Color SuperSamplingColor = Color(0.0f, 0.0f, 0.0f);
495  Point pixel1 = M + double(x) * pixelWidth + double(y) * pixelHeight; //subpixel "position"
496 
497  //Sampling the pixel with super sampling factor
498  for (int i = 0; i < camera.superSampling; i++) {
499  for (int j = 0; j < camera.superSampling; j++) {
500 
501  //offset to the subpixel center in wolrd coords
502  Point pixel((pixelHeight / double(camera.superSampling)) * double(i)
503  + (pixelWidth / double(camera.superSampling)) * double(j)
504  + (pixelWidth / (double(camera.superSampling) * double(camera.superSampling)))
505  + (pixelHeight / (double(camera.superSampling) * double(camera.superSampling))));
506 
507  //shifting to the subpixel center
508  pixel += pixel1;
509 
510  //We trace the ray
511  Ray ray(camera.eye, (pixel - camera.eye).normalized());
512  Color col = trace(ray, 0);
513 
514  //adding the color to the other samples
515  SuperSamplingColor += col;
516  }
517  }
518  //doing the mean of each subpixel for the current pixel
519  SuperSamplingColor = SuperSamplingColor / (double(camera.superSampling) * double(camera.superSampling));
520  SuperSamplingColor.clamp();
521  img(x, y) = SuperSamplingColor; //print to the image file
522  }
523  }
524 }
